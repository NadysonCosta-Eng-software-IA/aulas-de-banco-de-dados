CREATE TABLE FARMACIA 
( 
 nome_farmacia INT,  
 tel_farmacia INT,  
 CNPJ_farmacia INT PRIMARY KEY,  
 end_farmacia INT,  
); 

CREATE TABLE PRODUTO 
( 
 cod_produto INT PRIMARY KEY,  
 qtd_produto INT,  
 valor_produto INT,  
 idFARMACIA INT,  
); 

CREATE TABLE FAMACEUTICO 
( 
 CRF INT PRIMARY KEY,  
 nome_farmaceutico INT,  
 idFARMACIA INT,  
); 

ALTER TABLE PRODUTO ADD FOREIGN KEY(idFARMACIA) REFERENCES FARMACIA (idFARMACIA)
ALTER TABLE FAMACEUTICO ADD FOREIGN KEY(idFARMACIA) REFERENCES FARMACIA (idFARMACIA)
